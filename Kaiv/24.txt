import pandas as pd

# Load dataset
df = pd.read_csv("forestfires.csv")

# a) Create subsets
# Objective: Classify area affected

def classify(area):
    if area == 0:
        return "NotAffected"
    elif area <= 10:
        return "PartiallyAffected"
    else:
        return "MostlyAffected"

df['class'] = df['area'].apply(classify)

not_affected = df[df['class'] == 'NotAffected']
partial = df[df['class'] == 'PartiallyAffected']
mostly = df[df['class'] == 'MostlyAffected']

print(not_affected.head())

# b) Merge two subsets
# Objective: Combine two classes

merged = pd.concat([not_affected, partial])
print(merged.head())

# c) Sort data
# Objective: Sort by temperature, wind, and area

sorted_df = df.sort_values(by=['temp', 'wind', 'area'])
print(sorted_df.head())

# d) Transpose data
# Objective: Convert rows into columns

transpose = df.T
print(transpose.head())

# e) Melt data
# Objective: Convert wide format to long format

melted = pd.melt(df,
                 id_vars=['X', 'Y', 'month', 'day'],
                 var_name='Variable',
                 value_name='Value')

print(melted.head())

# f) Cast data to wide format
# Objective: Convert long format back to wide format

wide = melted.pivot_table(index=['X', 'Y', 'month', 'day'],
                          columns='Variable',
                          values='Value',
                          aggfunc='first').reset_index()

print(wide.head())
